import java.util.*;

 *  '1' - 문자 1

 */
public class Ex03_주민번호_나이 {
	public static void main(String[] args) {
		String id = "181212-3234567";
		String nai1 = id.substring(0, 2); // nai1 = "80" // substring은 문자열 자르기할때 쓰인다.
		int nai2 = Integer.parseInt(nai1); // String -> int("80"->80)
		int age = 0;

		// 1이거나 2라면 1900을
		// 3이거나 4라면 2000을
		if(sung == '1' || sung == '2') {
			age = year - (1900 + nai2) + 1;
		}else if(sung == '3' || sung == '4') {
			age = year - (2000 + nai2) + 1;
		}
		System.out.println("당신의 나이: " + age);

## for 개념
~~~java
/*
 * for(초기값;조건문;증가식)
 * {
 * 		반복할 문장들
 * }
 */
public class Ex05_for개념 {
	public static void main(String[] args) {
		// [1~10까지 합 구하기]
		int sum = 0;
		for(int i = 1; i <= 10; i++) {
			sum += i; // sum = sum + i;의 축약형
		}
		System.out.println("결과: " + sum);
		// [연습] 1~100까지 홀수의 합과 짝수의 합
		// [홀수의 합과 짝수의 합 변수]
		int sum1 = 0;
		int sum2 = 0;
		// [1~100까지 for문]
		for(int i = 1; i <= 100; i++) {
			// 짝수를 만드려면 2를 나눴을때 나머지가 0
			if(i % 2 == 0) {
				// 나머지가 0이면 sum1에 i를 더해준다.
				sum1 += i;
			}else {
				// 나머지가 0이 아니면 sum2에 i를 더해준다.
				sum2 += i;
			}
		}
		System.out.println("짝수의 합: " + sum1);
		System.out.println("홀수의 합: " + sum2);
		// [다른 방법의 홀수의 합과 짝수의 합]
		int sum1 = 0, sum2 = 0;
		for(int i = 1; i <= 100; i += 2) {
			sum1 += i;
			sum2 += (i + 1);
		}
		System.out.println("홀수의 합: " + sum1);
		System.out.println("짝수의 합: " + sum2);
		// [A~Z 출력]
		for(char c = 'A'; c <= 'Z'; c++) {
			System.out.print(c);
		}
		// [Z~A 출력]
		for(char d = 'Z'; d >= 'A'; d--) {
			System.out.print(d);
		}
	}
}
~~~

/* [3, 6, 9 게임]
 * 1~50까지 숫자 중에서 3, 6, 9가 포함되면 그 갯수만큼 '짝' 출력
 * 하나도 포함되지 않으면 그 숫자 출력

	public static void main(String[] args) {
		// [1~50까지 for문으로 만들기]
		for(int i = 1; i <= 50; i++) {
			// 변수 잡기;
			int mok = i;
			boolean su369 = false;
			// while에서 mok이 0이아닌 조건문을 만들어준다.
			while(mok != 0) {
				int na = mok % 10;
				// 10으로 나눈 mok의 나머지가 3, 6, 9이면 '짝'출력
				if(na == 3 | na == 6 | na == 9) {
					System.out.print("짝");
					su369 = true;
				}
				mok /= 10; 	
			}	
			if(su369) {
				System.out.println();
			}else {
				System.out.println(i);
			}
		}
		// [1~10000까지 8의 숫자 세기]
		// 8이 들어간 수를 합쳐주기 위한 변수
		int sum = 0;
		// 1~10000까지 for문 잡기
		for(int i = 1; i <= 10000; i++) {
			// 변수 잡기
			int mok = i;
			// mok이 0이 아니고 나머지8인 while문 잡기
			while(mok != 0) {
				int na = mok%10;
				if(na == 8) {
					sum += 1;
				}
				mok /= 10;
			}
		}
		System.out.println(sum);
	}
}
