import java.awt.*;
import java.awt.event.*;
import javax.swing.*;
import java.util.*;
class MyFrame extends JFrame {
	// 1. 멤버 변수 선언
	JButton bHamburger, bDessert, bDrink;
	JButton[][] kind = new JButton[3][3];
	JTextArea ta;
	int getsu = 3;
	public MyFrame() {
		// 2. 멤버 객체 선언
		new JFrame("버거킹");
		bHamburger = new JButton("햄버거", new ImageIcon("src/imgs/a.png"));
		bDessert = new JButton("디저트", new ImageIcon("src/imgs/b.png"));
		bDrink = new JButton("드링크", new ImageIcon("src/imgs/c.png"));
		ta = new JTextArea();
		for(int i = 0; i < kind.length; i++) {
			for(int j = 0; j < kind[i].length; j++) {
				kind[i][j] = new JButton();
			}
		}
	}
	// 3. 화면 붙이기 및 화면 출력
	void addLayout() {
		// north 영역
		JPanel p_north = new JPanel();
		p_north.setLayout(new GridLayout(1, 3));
		p_north.add(bHamburger);
		p_north.add(bDessert);
		p_north.add(bDrink);
		// center 영역
		JPanel p_center = new JPanel();
		p_center.setLayout(new GridLayout(getsu, getsu));
		for(int i = 0; i < kind.length; i++) {
			for(int j = 0; j < kind[i].length; j++) {
				p_center.add(kind[i][j]);
			}
		}
		// south 영역
		JPanel p_south = new JPanel();
		p_south.add(ta);
		// 프레임 영역
		setLayout(new BorderLayout());
		add(p_north, BorderLayout.NORTH);
		add(p_center, BorderLayout.CENTER);
		add(p_south, BorderLayout.SOUTH);
		setSize(900, 500);
		setVisible(true);
		setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
		
	}
	// 4. 이벤트 등록 및 처리
	void eventProc() {
		bHdlr b = new bHdlr();
		bHamburger.addActionListener(b);
		bDessert.addActionListener(b);
		bDrink.addActionListener(b);
	}
	class bHdlr implements ActionListener{
		public void actionPerformed(ActionEvent e) {
			Object obj = e.getSource();
			if(obj == bHamburger) {
				hamburger();
			}else if(obj == bDessert) {
				dessert();
			}else if(obj == bDrink) {
				drink();
			}
		}
	}
	void hamburger() {
		
	}
	void dessert() {
		
	}
	void drink() {
		
	}
}
public class Test {
	public static void main(String[] args) {
		MyFrame f = new MyFrame();
		f.addLayout();
		f.eventProc();
	}
}
